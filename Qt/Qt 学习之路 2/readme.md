import os
import sys
import re

def getTextPath():
    return "/home/xxxx/Desktop/url.txt"

def readFile(file):
    with open(file) as f:
        data = f.read()
    f.close()

    return data

# www.bookstack.cn/read/qt-study-road-2/b173882cafb6bb28.md
def findUrl(text):
    url_pattern = r'www.(.*?).md'
    url_list = re.findall(url_pattern, text, flags = 0)

    return url_list

def findNameFromUrl(text):
    name_pattern = r'title=\"(.*?)\"'
    name_list = re.findall(name_pattern, text, flags = 0)

    return name_list

if __name__ == '__main__':
    text = readFile(getTextPath())
    url_list = findUrl(text)
    name_list = findNameFromUrl(text)

    i = 0
    f = open("/home/xxxxx/Desktop/res.txt", "w+")
    for i in range(len(url_list)): 
        download_cmd = "wget www." + url_list[i] + ".md -O \"" + str(i) + "_" + name_list[i] + ".html\""
        os.system(download_cmd)
        print(download_cmd)
        i += 1
    f.close()
